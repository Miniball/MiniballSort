578void MiniballEventBuilder::GammaRayFinder() {
579
580 // Temporary variables for addback
581 unsigned long long MaxTime; // time of event with maximum energy
582 unsigned char MaxSegId; // segment with maximum energy
583 unsigned char MaxCryId; // crystal with maximum energy
584 float MaxEnergy; // maximum core energy
585 float MaxSegEnergy; // maximum segment energy
586 float SegSumEnergy; // add segment energies
587 float AbSumEnergy; // add core energies for addback
588 unsigned char seg_mul; // segment multiplicity
589 unsigned char ab_mul; // addback multiplicity
590 std::vector<unsigned char> ab_index; // index of addback already used
591 bool skip_event; // has this event been used already
592
593 // Loop over all the events in Miniball detectors
594 for( unsigned int i = 0; i < mb_en_list.size(); ++i ) {
595
596 // Check if it's a core event
597 if( mb_seg_list.at(i) != 0 ) continue;
598
599 // Reset addback variables
600 MaxSegId = 0; // initialise as core (if no segment hit (dead), use core!)
601 MaxSegEnergy = 0.;
602 SegSumEnergy = 0.;
603 seg_mul = 0;
604
605 // Loop again to find the matching segments
606 for( unsigned int j = 0; j < mb_en_list.size(); ++j ) {
607
608 // Skip the same event
609 if( i == j ) continue;
610
611 // Skip if it's a core again, also fill time diff plot
612 if( mb_seg_list.at(j) == 0 ) {
613
614 // Fill the time difference spectrum
615 mb_td_core_core->Fill( (long long)mb_ts_list.at(i) - (long long)mb_ts_list.at(j) );
616 continue;
617
618 }
619
620 // Skip if it's not the same crystal and cluster
621 if( mb_clu_list.at(i) != mb_clu_list.at(j) ||
622 mb_cry_list.at(i) != mb_cry_list.at(j) ) continue;
623
624 // Fill the time difference spectrum
625 mb_td_core_seg->Fill( (long long)mb_ts_list.at(i) - (long long)mb_ts_list.at(j) );
626
627 // Skip if we are outside of the hit window
628 if( TMath::Abs( (double)mb_ts_list.at(i) - (double)mb_ts_list.at(j) )
629 > set->GetMiniballCrystalHitWindow() ) continue;
630
631 // Increment the segment multiplicity and sum energy
632 seg_mul++;
633 SegSumEnergy += mb_en_list.at(j);
634
635 // Is this bigger than the current maximum energy?
636 if( mb_en_list.at(j) > MaxSegEnergy ){
637
638 MaxSegEnergy = mb_en_list.at(j);
639 MaxSegId = mb_seg_list.at(j);
640
641 }
642
643 } // j: matching segments
644
645 // Fill the segment spectra with core energies
646 mb_en_core_seg[mb_clu_list.at(i)][mb_cry_list.at(i)]->Fill( MaxSegId, mb_en_list.at(i) );
647 if( mb_ts_list.at(i) - ebis_time < 1.5e6 )
648 mb_en_core_seg_ebis_on[mb_clu_list.at(i)][mb_cry_list.at(i)]->Fill( MaxSegId, mb_en_list.at(i) );
649
650
651 //if( MaxSegId == 0 && mb_en_list.at(i) > 150.0 )
652 // std::cout << std::endl << mb_en_list.at(i) << "\t" << MaxSegEnergy << std::endl;
653
654 // Build the single crystal gamma-ray event
655 gamma_ctr++;
656 gamma_evt->SetEnergy( mb_en_list.at(i) );
657 gamma_evt->SetSegmentEnergy( MaxSegEnergy );
658 gamma_evt->SetCluster( mb_clu_list.at(i) );
659 gamma_evt->SetCrystal( mb_cry_list.at(i) );
660 gamma_evt->SetSegment( MaxSegId );
661 gamma_evt->SetTime( mb_ts_list.at(i) );
662 write_evts->AddEvt( gamma_evt );
663
664 } // i: core events
665
666
667 // Loop over all the gamma-ray singles for addback
668 for( unsigned int i = 0; i < write_evts->GetGammaRayMultiplicity(); ++i ) {
669
670 // Reset addback variables
671 AbSumEnergy = write_evts->GetGammaRayEvt(i)->GetEnergy();
672 MaxCryId = write_evts->GetGammaRayEvt(i)->GetCrystal();
673 MaxSegId = write_evts->GetGammaRayEvt(i)->GetSegment();
674 MaxEnergy = AbSumEnergy;
675 MaxSegEnergy = write_evts->GetGammaRayEvt(i)->GetSegmentEnergy();
676 MaxTime = write_evts->GetGammaRayEvt(i)->GetTime();
677 ab_mul = 1; // this is already the first event
678
679 // Check we haven't already used this event
680 skip_event = false;
681 for( unsigned int k = 0; k < ab_index.size(); ++k )
682 if( ab_index.at(k) == i ) skip_event = true;
683 if( skip_event ) continue;
684
685 // Loop to find a matching event for addback
686 for( unsigned int j = i+1; j < write_evts->GetGammaRayMultiplicity(); ++j ) {
687
688 // Make sure we are in the same cluster
689 // In the future we might consider a more intelligent
690 // algorithm, which uses the line-of-sight idea
691 if( write_evts->GetGammaRayEvt(i)->GetCluster() !=
692 write_evts->GetGammaRayEvt(j)->GetCluster() ) continue;
693
694 // Skip if we are outside of the hit window
695 if( TMath::Abs( (double)write_evts->GetGammaRayEvt(i)->GetTime() -
696 (double)write_evts->GetGammaRayEvt(j)->GetTime() )
697 > set->GetMiniballAddbackHitWindow() ) continue;
698
699 // Check we haven't already used this event
700 skip_event = false;
701 for( unsigned int k = 0; k < ab_index.size(); ++k )
702 if( ab_index.at(k) == j ) skip_event = true;
703 if( skip_event ) continue;
704
705 // Then we can add them back
706 ab_mul++;
707 AbSumEnergy += write_evts->GetGammaRayEvt(j)->GetEnergy();
708 ab_index.push_back(j);
709
710 // Is this bigger than the current maximum energy?
711 if( write_evts->GetGammaRayEvt(j)->GetEnergy() > MaxEnergy ){
712
713 MaxEnergy = write_evts->GetGammaRayEvt(j)->GetEnergy();
714 MaxSegEnergy = write_evts->GetGammaRayEvt(j)->GetSegmentEnergy();
715 MaxCryId = write_evts->GetGammaRayEvt(j)->GetCrystal();
716 MaxSegId = write_evts->GetGammaRayEvt(j)->GetSegment();
717 MaxTime = write_evts->GetGammaRayEvt(j)->GetTime();
718
719 }
720
721 } // j: loop for matching addback
722
723 // Build the single crystal gamma-ray event
724 gamma_ab_ctr++;
725 gamma_ab_evt->SetEnergy( AbSumEnergy );
726 gamma_ab_evt->SetSegmentEnergy( MaxSegEnergy );
727 gamma_ab_evt->SetCluster( write_evts->GetGammaRayEvt(i)->GetCluster() );
728 gamma_ab_evt->SetCrystal( MaxCryId );
729 gamma_ab_evt->SetSegment( MaxSegId );
730 gamma_ab_evt->SetTime( MaxTime );
731 write_evts->AddEvt( gamma_ab_evt );
732
733 } // i: gamma-ray singles
734
735 return;
736
737}
