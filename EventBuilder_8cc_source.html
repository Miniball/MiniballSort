559void MiniballEventBuilder::GammaRayFinder() {
560
561 // Temporary variables for addback
562 unsigned long long MaxTime; // time of event with maximum energy
563 unsigned char MaxSegId; // segment with maximum energy
564 unsigned char MaxCryId; // crystal with maximum energy
565 float MaxEnergy; // maximum core energy
566 float MaxSegEnergy; // maximum segment energy
567 float SegSumEnergy; // add segment energies
568 float AbSumEnergy; // add core energies for addback
569 unsigned char seg_mul; // segment multiplicity
570 unsigned char ab_mul; // addback multiplicity
571 std::vector<unsigned char> ab_index; // index of addback already used
572 bool skip_event; // has this event been used already
573
574 // Loop over all the events in Miniball detectors
575 for( unsigned int i = 0; i < mb_en_list.size(); ++i ) {
576
577 // Check if it's a core event
578 if( mb_seg_list.at(i) != 0 ) continue;
579
580 // Reset addback variables
581 MaxSegId = 0; // initialise as core (if no segment hit (dead), use core!)
582 MaxSegEnergy = 0.;
583 SegSumEnergy = 0.;
584 seg_mul = 0;
585
586 // Loop again to find the matching segments
587 for( unsigned int j = 0; j < mb_en_list.size(); ++j ) {
588
589 // Skip the same event
590 if( i == j ) continue;
591
592 // Skip if it's a core again, also fill time diff plot
593 if( mb_seg_list.at(j) == 0 ) {
594
595 // Fill the time difference spectrum
596 mb_td_core_core->Fill( (long long)mb_ts_list.at(i) - (long long)mb_ts_list.at(j) );
597 continue;
598
599 }
600
601 // Skip if it's not the same crystal and cluster
602 if( mb_clu_list.at(i) != mb_clu_list.at(j) ||
603 mb_cry_list.at(i) != mb_cry_list.at(j) ) continue;
604
605 // Fill the time difference spectrum
606 mb_td_core_seg->Fill( (long long)mb_ts_list.at(i) - (long long)mb_ts_list.at(j) );
607
608 // Skip if we are outside of the hit window
609 if( TMath::Abs( (double)mb_ts_list.at(i) - (double)mb_ts_list.at(j) )
610 > set->GetMiniballCrystalHitWindow() ) continue;
611
612 // Increment the segment multiplicity and sum energy
613 seg_mul++;
614 SegSumEnergy += mb_en_list.at(j);
615
616 // Is this bigger than the current maximum energy?
617 if( mb_en_list.at(j) > MaxSegEnergy ){
618
619 MaxSegEnergy = mb_en_list.at(j);
620 MaxSegId = mb_seg_list.at(j);
621
622 }
623
624 } // j: matching segments
625
626 // Fill the segment spectra with core energies
627 mb_en_core_seg[mb_clu_list.at(i)][mb_cry_list.at(i)]->Fill( MaxSegId, mb_en_list.at(i) );
628 if( mb_ts_list.at(i) - ebis_time < 1.5e6 )
629 mb_en_core_seg_ebis_on[mb_clu_list.at(i)][mb_cry_list.at(i)]->Fill( MaxSegId, mb_en_list.at(i) );
630
631
632 //if( MaxSegId == 0 && mb_en_list.at(i) > 150.0 )
633 // std::cout << std::endl << mb_en_list.at(i) << "\t" << MaxSegEnergy << std::endl;
634
635 // Build the single crystal gamma-ray event
636 gamma_ctr++;
637 gamma_evt->SetEnergy( mb_en_list.at(i) );
638 gamma_evt->SetSegmentEnergy( MaxSegEnergy );
639 gamma_evt->SetCluster( mb_clu_list.at(i) );
640 gamma_evt->SetCrystal( mb_cry_list.at(i) );
641 gamma_evt->SetSegment( MaxSegId );
642 gamma_evt->SetTime( mb_ts_list.at(i) );
643 write_evts->AddEvt( gamma_evt );
644
645 } // i: core events
646
647
648 // Loop over all the gamma-ray singles for addback
649 for( unsigned int i = 0; i < write_evts->GetGammaRayMultiplicity(); ++i ) {
650
651 // Reset addback variables
652 AbSumEnergy = write_evts->GetGammaRayEvt(i)->GetEnergy();
653 MaxCryId = write_evts->GetGammaRayEvt(i)->GetCrystal();
654 MaxSegId = write_evts->GetGammaRayEvt(i)->GetSegment();
655 MaxEnergy = AbSumEnergy;
656 MaxSegEnergy = write_evts->GetGammaRayEvt(i)->GetSegmentEnergy();
657 MaxTime = write_evts->GetGammaRayEvt(i)->GetTime();
658 ab_mul = 1; // this is already the first event
659
660 // Check we haven't already used this event
661 skip_event = false;
662 for( unsigned int k = 0; k < ab_index.size(); ++k )
663 if( ab_index.at(k) == i ) skip_event = true;
664 if( skip_event ) continue;
665
666 // Loop to find a matching event for addback
667 for( unsigned int j = i+1; j < write_evts->GetGammaRayMultiplicity(); ++j ) {
668
669 // Make sure we are in the same cluster
670 // In the future we might consider a more intelligent
671 // algorithm, which uses the line-of-sight idea
672 if( write_evts->GetGammaRayEvt(i)->GetCluster() !=
673 write_evts->GetGammaRayEvt(j)->GetCluster() ) continue;
674
675 // Skip if we are outside of the hit window
676 if( TMath::Abs( (double)write_evts->GetGammaRayEvt(i)->GetTime() -
677 (double)write_evts->GetGammaRayEvt(j)->GetTime() )
678 > set->GetMiniballAddbackHitWindow() ) continue;
679
680 // Check we haven't already used this event
681 skip_event = false;
682 for( unsigned int k = 0; k < ab_index.size(); ++k )
683 if( ab_index.at(k) == j ) skip_event = true;
684 if( skip_event ) continue;
685
686 // Then we can add them back
687 ab_mul++;
688 AbSumEnergy += write_evts->GetGammaRayEvt(j)->GetEnergy();
689 ab_index.push_back(j);
690
691 // Is this bigger than the current maximum energy?
692 if( write_evts->GetGammaRayEvt(j)->GetEnergy() > MaxEnergy ){
693
694 MaxEnergy = write_evts->GetGammaRayEvt(j)->GetEnergy();
695 MaxSegEnergy = write_evts->GetGammaRayEvt(j)->GetSegmentEnergy();
696 MaxCryId = write_evts->GetGammaRayEvt(j)->GetCrystal();
697 MaxSegId = write_evts->GetGammaRayEvt(j)->GetSegment();
698 MaxTime = write_evts->GetGammaRayEvt(j)->GetTime();
699
700 }
701
702 } // j: loop for matching addback
703
704 // Build the single crystal gamma-ray event
705 gamma_ab_ctr++;
706 gamma_ab_evt->SetEnergy( AbSumEnergy );
707 gamma_ab_evt->SetSegmentEnergy( MaxSegEnergy );
708 gamma_ab_evt->SetCluster( write_evts->GetGammaRayEvt(i)->GetCluster() );
709 gamma_ab_evt->SetCrystal( MaxCryId );
710 gamma_ab_evt->SetSegment( MaxSegId );
711 gamma_ab_evt->SetTime( MaxTime );
712 write_evts->AddEvt( gamma_ab_evt );
713
714 } // i: gamma-ray singles
715
716 return;
717
718}
