447TVector3 MiniballReaction::GetCDVector( unsigned char det, unsigned char sec, float pid, float nid ){
448
449 // Check that we have a real CD detector
450 if( det >= set->GetNumberOfCDDetectors() ) {
451
452 std::cerr << "Bad CD Detector requested = " << det << std::endl;
453 det = 0;
454
455 }
456
457 // Create a TVector3 to handle the angles
458 TVector3 vec( 0, 0, cd_dist[det] ); // set z now
459
460 // Get the phi rotation of the quadrant
461 float phi = 90.0 * sec;
462 phi += cd_offset[det]; // left edge of first strip
463
464 // Recalculate this points for the standard CD (not yet done for CREX/TREX)
465 if( set->GetNumberOfCDNStrips() == 12 ) { // standard CD
466
467 // CD phi calculation using method from Tim Gray
468 double alpha = 82.0;
469 double offset = 1.625;
470
471 double phi_body = ( 2.0*nid + 1.0 ) * alpha / 24.;
472
473 double r_lab = 9.0;
474 r_lab += ( 15.5 - pid ) * 2.0; // pid = 0 is outer ring and pid = 15 is inner ring
475
476 double beta = 180.0 - alpha / 2.0;
477 double bphi = beta + phi_body;
478 if( bphi > 180.0) bphi = 360.0 - bphi;
479
480 double r_d = offset / TMath::Sin( alpha * TMath::DegToRad() / 2.0 ); // from center of rings to center of sectors
481 double delta = TMath::ASin( r_d * TMath::Sin( bphi * TMath::DegToRad() ) / r_lab );
482 delta *= TMath::RadToDeg(); // angle between r_body and r_lab
483
484 double gamma = 180.0 - bphi - delta; // angle between r_d and r_lab
485
486 double r_body = TMath::Sin( gamma * TMath::DegToRad() ) / ( TMath::Sin( bphi * TMath::DegToRad() ) / r_lab ); // between sector center and point of interest
487
488 double x_body = r_body * TMath::Cos( phi_body * TMath::DegToRad() ); //in sector "body" coordinates
489 double y_body = r_body * TMath::Sin( phi_body * TMath::DegToRad() );
490
491 //transform back to ring "lab" coordinates
492 double y = y_body + offset;
493 double x = x_body + TMath::Sqrt( r_d*r_d - offset*offset );
494
495 //should have sqrt(x*x + y*y) = r_lab at this point
496 vec.SetX(x);
497 vec.SetY(y);
498
499 /*
500 // New definition of CD segments - Konstantin Stoychev
502 //float initial_offset = 0.000286 * TMath::Power(pid,4);
503 //initial_offset += -0.00541 * TMath::Power(pid,3);
504 //initial_offset += 0.04437 * TMath::Power(pid,2);
505 //initial_offset += 0.01679 * pid;
506 //initial_offset += 2.4137;
