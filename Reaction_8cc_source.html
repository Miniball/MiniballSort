445TVector3 MiniballReaction::GetCDVector( unsigned char det, unsigned char sec, float pid, float nid ){
446
447 // Check that we have a real CD detector
448 if( det >= set->GetNumberOfCDDetectors() ) {
449
450 std::cerr << "Bad CD Detector requested = " << det << std::endl;
451 det = 0;
452
453 }
454
455 // Create a TVector3 to handle the angles
456 TVector3 vec( 0, 0, cd_dist[det] ); // set z now
457
458 // Get the phi rotation of the quadrant
459 float phi = 90.0 * sec;
460 phi += cd_offset[det]; // left edge of first strip
461
462 // Recalculate this points for the standard CD (not yet done for CREX/TREX)
463 if( set->GetNumberOfCDNStrips() == 12 || set->GetNumberOfCDNStrips() == 24 ) { // standard CD
464
465 // CD phi calculation using method from Tim Gray
466 double alpha = 82.0;
467 double offset = 1.625;
468 double grouping = 24.0 / (double)set->GetNumberOfCDNStrips();
469
470 double phi_body = ( grouping * nid + 1.0 ) * alpha / 24.;
471
472 double r_lab = 9.0;
473 r_lab += ( 15.5 - pid ) * 2.0; // pid = 0 is outer ring and pid = 15 is inner ring
474
475 double beta = 180.0 - alpha / 2.0;
476 double bphi = beta + phi_body;
477 if( bphi > 180.0) bphi = 360.0 - bphi;
478
479 double r_d = offset / TMath::Sin( alpha * TMath::DegToRad() / 2.0 ); // from center of rings to center of sectors
480 double delta = TMath::ASin( r_d * TMath::Sin( bphi * TMath::DegToRad() ) / r_lab );
481 delta *= TMath::RadToDeg(); // angle between r_body and r_lab
482
483 double gamma = 180.0 - bphi - delta; // angle between r_d and r_lab
484
485 double r_body = TMath::Sin( gamma * TMath::DegToRad() ) / ( TMath::Sin( bphi * TMath::DegToRad() ) / r_lab ); // between sector center and point of interest
486
487 double x_body = r_body * TMath::Cos( phi_body * TMath::DegToRad() ); //in sector "body" coordinates
488 double y_body = r_body * TMath::Sin( phi_body * TMath::DegToRad() );
489
