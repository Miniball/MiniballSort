78// Treat a Mesytec ADC data item
79void MiniballMedConverter::ProcessMesytecAdcData() {
80
81 // Loop over all the available data
82 unsigned int i = 0;
83 while( i < mbs_sevt->GetNumberOfData() ){
84
85 // Header of the sub event (module number in first word)
86 unsigned short header = mbs_sevt->GetData(i++);
87 unsigned short mod = header & MESYTEC_MADC_MODULE_ID;
88
89 // Convert to logical module from Marabou numbering
90 if( mod >= set->GetNumberOfAdcModules() )
91 mod = set->GetMesytecAdcModuleNumber( mod );
92 else mod--; // Mesytec modules count from 1 in Marabou!!
93
94 // Get second word of the header and test format
95 header = mbs_sevt->GetData(i++);
96 if( header & MESYTEC_MADC_OUTPUT_FORMAT ) {
97
98 std::cerr << __PRETTY_FUNCTION__ << ": Error, output ";
99 std::cerr << "format (highest bit) should be zero (header = ";
100 std::cerr << std::hex << header << std::dec << ")" << std::endl;
101 return;
102
103 }
104
105 // If format is fine, we have the word count in here
106 int wc = (header & MESYTEC_MADC_WORD_COUNT);
107
108 // Is module number ok? If not, get next item
109 if( mod < 0 ) {
110 i += wc * 2; // skip rest of event (2 shorts per word)
111 continue;
112 }
113
114 // check number of Channels (MESYTEC_MADC_NBOFCHAN=32 + 1 word End of Event + 1 extended timestamp)
115 // -> skip TOTAL subevent if wrong number of Channels
116 if( wc <= 0 || wc > (int)set->GetNumberOfMesytecAdcChannels() + 2 ) {
117
118 std::cout << __PRETTY_FUNCTION__ << ": read event nr. ";
119 std::cout << my_event_id << ": wrong Word Count: ";
120 std::cout << wc << " -> skip TOTAL subevent" << std::endl;
121 return;
122
123 }
124
125 // Loop over number of channels for which data follows (maybe also extended timestamp)
126 // Might not be all channel, so use wc, but wc includes the end of event
127 std::vector<unsigned short> ch_vec;
128 std::vector<unsigned short> qshort_vec;
129 long long Timestamp = 0;
130 bool clipped = false;
131 for( unsigned short ch = 0; ch < wc - 1; ch++ ) {
132
133 // Check if type of word (highest two bits) is end of event
134 unsigned short test = mbs_sevt->GetData(i++);
135
136 // 00 => data, 11 => end of event
137 if( ( test & MESYTEC_MADC_END_OF_EVENT ) == MESYTEC_MADC_END_OF_EVENT ) {
138
139 std::cerr << "Error, found end of event data (" << test << ") after " << ch+1;
140 std::cerr << " words but should have a word count of " << wc << std::endl;
141 return;
142
143 }
144
145 // What if only one of the higest two bits is non-zero? weird
146 else if( test & MESYTEC_MADC_END_OF_EVENT ) {
147
148 std::cerr << "Error, found weird data (" << test << ") after ";
149 std::cerr << ch+1 << " words with word count of " << wc << std::endl;
150 return;
151
152 }
153
154 // Check whether this is the extended timestamp
155 if( test & MESYTEC_MADC_EXTENDED_TIMESTAMP ) {
156
157 Timestamp = ((long long)mbs_sevt->GetData(i++)) << MESYTEC_MADC_EXTENDED_TIMESTAMP_SHIFT;
158 //Timestamp &= MESYTEC_MADC_EXTENDED_TIMESTAMP_MASK;
159 continue;
160
161 }
162
163 // If we don't have the timestamp here, test has the channel number
164 // Get actual Channel number (bits[21...16] of data word)
165 unsigned short chanNo = (test & MESYTEC_MADC_CHANNEL_NUMBER);
166 ch_vec.push_back( chanNo );
167
168 // But the energy is in the next word, lowest 12 bits
169 unsigned short Qshort = (mbs_sevt->GetData(i++) & MESYTEC_MADC_VALUE);
170 qshort_vec.push_back( Qshort );
171
172 // Check the out of range bit
173 if( mbs_sevt->GetData(i) & MESYTEC_MADC_OUT_OF_RANGE )
174 clipped = true;
175 else clipped = false;
176
177 } // loop over wc - 1
178
179 // Trailer of the sub event (module number in first word)
180 unsigned int trailer = ((unsigned int)mbs_sevt->GetData(i++) << 16) & 0xffff0000;
181 trailer |= ((unsigned int)mbs_sevt->GetData(i++)) & 0x0000ffff;
182
183 // Check type of word (highest two bits should be set)
184 if ( ( trailer & MESYTEC_MADC_END_OF_EVENT ) != MESYTEC_MADC_END_OF_EVENT ) {
185
186 std::cout << __PRETTY_FUNCTION__ << ": read event nr. " << my_event_id;
187 std::cout << ": wrong EOE word of type: " << ( trailer & MESYTEC_MADC_END_OF_EVENT );
188 std::cout << " -> skip TOTAL subevent" << std::endl;
189 return;
190
191 }
192
193 // Merge the full time stamp and add the DGF delay
194 Timestamp |= ( trailer & MESYTEC_MADC_TIMESTAMP );
195 Timestamp += set->GetDgfTimestampDelay();
196 Timestamp *= set->GetMesytecAdcTimestampUnits();
197
198 // Now we have the data, fill the tree
199 for( unsigned item = 0; item < qshort_vec.size(); item++ ){
200
201 // Clear the old stuff
202 adc_data->ClearData();
203 //data_packet->ClearData();
204
205 // Some basic info for every event
206 adc_data->SetEventID( my_event_id );
207
208 // Calculate energy and threshold
209 float energy = cal->AdcEnergy( mod, ch_vec[item], qshort_vec[item] );
210 bool thresh = cal->AdcThreshold( mod, ch_vec[item] );
211
212 // Corrected time for ADCs
213 long long time_corr = Timestamp;
214 time_corr += cal->AdcTime( mod, ch_vec[item] );
215
216 // Set values for data item
217 adc_data->SetTime( time_corr ); // only works for MADC, CAEN needs reconstruction
218 adc_data->SetQshort( qshort_vec[item] );
219 adc_data->SetModule( mod );
220 adc_data->SetChannel( (char)ch_vec[item] );
221 adc_data->SetEnergy( energy );
222 adc_data->SetThreshold( thresh );
223 adc_data->SetClipped( clipped );
224
225 // Fill the tree
226 if( !flag_source ) {
227 std::shared_ptr<MiniballDataPackets> data_packet =
228 std::make_shared<MiniballDataPackets>( adc_data );
229 data_vector.emplace_back( data_packet );
230 data_map.push_back( std::make_pair<unsigned long,double>(
231 data_vector.size()-1, data_packet->GetTime() ) );
232 }
233
234 // Fill histograms
235 hadc_qshort[mod][ch_vec[item]]->Fill( qshort_vec[item] );
236 hadc_cal[mod][ch_vec[item]]->Fill( energy );
237
238 }
239
240 }
241
242 return;
243
244}

360// Treat a DGF scaler data item
361void MiniballMedConverter::ProcessDgfScaler() {
362
363 // Test data comes from DGFscaler unit?
364 if( !set->CheckVmeModuleIsDgfScaler( mbs_sevt->GetModuleID() ) ){
365
366 std::cerr << __PRETTY_FUNCTION__;
367 std::cerr << ": Error - got data from VME module id ";
368 std::cerr << mbs_sevt->GetModuleID();
369 std::cerr << ", which looks like a DGF scaler module but it isn't";
370 std::cerr << " defined as such in settings file" << std::endl;
371 return;
372
373 }
374
375 // Loop over all the available data
376 unsigned int i = 0;
377 while( i < mbs_sevt->GetNumberOfData() ){
378
379 // Finish the data?
380 if( mbs_sevt->GetData(i) == DGF_SCALER_END_OF_BUFFER )
381 break;
382
383 // Check for magic word at the start
384 if( mbs_sevt->GetData(i++) != DGF_SCALER_MAGIC_WORD ){
385
386 std::cerr << "Internal dgf scalers: data out of phase - 0x";
387 std::cerr << std::hex << mbs_sevt->GetData(--i);
388 std::cerr << " (should be magic word 0x" << DGF_SCALER_MAGIC_WORD ;
389 std::cerr << ")" << std::dec << std::endl;
390 return;
391
392 }
393
394 // Get the header data
395 int wc = mbs_sevt->GetData(i++); // word count
396 int clu = mbs_sevt->GetData(i++); // cluster id
397 int mod = mbs_sevt->GetData(i++); // module id
398
399 // Break if we have nonsense
400 if( clu < 0 || mod < 0 ) return;
401 if( wc < 0 ){
402 std::cerr << "Word count negative: " << wc << std::endl;
403 return;
404 }
405 if( (int)mbs_sevt->GetNumberOfData() - (int)(i-2) < DGF_SCALER_MIN_SIZE ){
406 std::cerr << "Not enough data left in sub event for new DGF scaler: ";
407 std::cerr << mbs_sevt->GetNumberOfData() - (i-2) << " < minsize(";
408 std::cerr << DGF_SCALER_MIN_SIZE << ")" << std::endl;
409 return;
410 }
411 if( (int)mbs_sevt->GetNumberOfData() - (int)(i-2) < wc ){
412 std::cerr << "Not enough data left in sub event for new DGF scaler: ";
413 std::cerr << mbs_sevt->GetNumberOfData() - (i-2) << " < wc(" << wc << ")" << std::endl;
414 return;
415 }
416
417 // Get the real time data
418 unsigned int index = i + DGF_SCALER_INDEX_REALTIME;
419 long long realTime = mbs_sevt->GetData(index);
420 realTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
421 realTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
422
423 // Get the run time data
424 index = i + DGF_SCALER_INDEX_RUNTIME;
425 long long runTime = mbs_sevt->GetData(index);
426 runTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
427 runTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
428
429 // Get the GSLT time data
430 index = i + DGF_SCALER_INDEX_GSLTTIME;
431 long long gsltTime = mbs_sevt->GetData(index);
432 gsltTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
433 gsltTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
434
435 // Get the number of events
436 index = i + DGF_SCALER_INDEX_NEVENTS;
437 long long nEvents = mbs_sevt->GetData(index);
438 nEvents |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
439 nEvents |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
440
441 // Make a DGF scaler event and add it to the packets
442 DgfScalerData s( set->GetNumberOfDgfChannels() );
443 s.SetModule( mod );
444 s.SetClusterID( clu );
445 s.SetRealTime( realTime );
446 s.SetRunTime( runTime );
447 s.SetGSLTTime( gsltTime );
448
449 // Get channel by channel data
450 for( unsigned int j = 0; j < set->GetNumberOfDgfChannels(); j++ ){
451
452 index = i + DGF_SCALER_INDEX_CH_OFFSET;
453 index += j * DGF_SCALER_INDEX_CH_SIZE;
454
455 // Get LiveTime for this channel
456 long long liveTime = mbs_sevt->GetData(index);
457 liveTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
458 liveTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
459
460 // Get FastPeak for this channel
461 int fastPeak = mbs_sevt->GetData(index+3);
462 fastPeak |= ( (unsigned long long)mbs_sevt->GetData(index+4) ) << 16;
463
464 s.SetLiveTime( j, liveTime );
465 s.SetFastPeak( j, fastPeak );
466
467 }
468
469 // Move forward to next data item
470 i += wc - 4; // 4 header words included in word count
471
472 }
473
474 return;
475

531 ( format != COMP_LM_BUFFORMAT ) &&

613 for( unsigned int ch = 0; ch < set->GetNumberOfDgfChannels(); ch++ ) {

621 //std::cout << std::dec << std::endl;

627 ChannelLength = mbs_sevt->GetData(i++);

630 unsigned short FastTriggerTime = mbs_sevt->GetData(i++);
631 unsigned short Qshort = mbs_sevt->GetData(i++);
632
633 // Sort out long fast trigger time and wrap around
634 unsigned long long LongFastTriggerTime = FastTriggerTime;
635 if( FastTriggerTime > EventTimeLow )
636 LongFastTriggerTime += 65536ll*EventTimeHigh + 65536ll*65536ll*RunTimeA;
637 else
638 LongFastTriggerTime += 65536ll*EventTimeHigh + 65536ll + 65536ll*65536ll*RunTimeA;
639
640 // Get calibrated energy and check threshold
641 float energy = cal->DgfEnergy( mod, ch, Qshort );
642 bool thresh = cal->DgfThreshold( mod, ch );
643
644 // For RUNTASK!=259, now 6 user PSA values (& possible trace) follow
645 std::vector<unsigned short> UserValues;
646 std::vector<unsigned short> trace;
647 if( format != COMP3_LM_BUFFORMAT ) {
648
649 // Get 6 user values for PSA
650 for( unsigned char j = 0; j < 6; j++ )
651 UserValues.push_back( mbs_sevt->GetData(i++) );
652
653 // Read out trace
654 unsigned int TraceLength = (int)ChannelLength - CHANHEADLEN;
655 for( unsigned char j = 0; j < TraceLength; j++ )
656 trace.push_back( mbs_sevt->GetData(i++) );
657
658 } // check format for PSA and trace
659
660 //-----------
661 // Now let's add the data
662 //------------
663
664 // Fill histograms
665 hdgf_qshort[mod][ch]->Fill( Qshort );
666 hdgf_cal[mod][ch]->Fill( energy );
667
668 // Clear the old stuff
669 dgf_data->ClearData();
670 info_data->ClearData();
671 //data_packet->ClearData();
672
673 //std::cout << "DGF module " << mod << ", time = ";
674 //std::cout << LongFastTriggerTime << ", Qshort = ";
675 //std::cout << Qshort << std::endl;
676
677 long long time_corr = LongFastTriggerTime * set->GetDgfTimestampUnits();
678 time_corr += cal->DgfTime( mod, ch );
679
680 // Check if it's a timestamper!
681 if( set->IsTimestampModule( mod ) ) {
682
683 // Check if it matches EBIS, T1 or SC
684 char mycode = -1;
685 if( mod == set->GetEBISDgf() && ch == set->GetEBISChannel() )
686 mycode = set->GetEBISCode();
687 else if( mod == set->GetT1Dgf() && ch == set->GetT1Channel() )
688 mycode = set->GetT1Code();
689 else if( mod == set->GetSCDgf() && ch == set->GetSCChannel() )
690 mycode = set->GetSCCode();
691
692 // Useful only if it matched
693 if( mycode >= 0 ) {
694
695 // Set values for data item
696 info_data->SetEventID( my_event_id );
697 info_data->SetTime( time_corr );
698 info_data->SetCode( mycode );
699 info_data->SetBoard( mod );
700
701 // Fill the tree
702 if( !flag_source ) {
703 std::shared_ptr<MiniballDataPackets> data_packet =
704 std::make_shared<MiniballDataPackets>( info_data );
705 data_vector.emplace_back( data_packet );
706 data_map.push_back( std::make_pair<unsigned long,double>(
707 data_vector.size()-1, data_packet->GetTime() ) );
708 }
709
710 }
711
712 }
713
714 else {
715
716 // Set values for data item
717 dgf_data->SetEventID( my_event_id );
718 dgf_data->SetRunTime( RunTime * set->GetDgfTimestampUnits() );
719 dgf_data->SetEventTime( EventTime * set->GetDgfTimestampUnits() );
720 dgf_data->SetFastTriggerTime( FastTriggerTime * set->GetDgfTimestampUnits() );
721 dgf_data->SetLongFastTriggerTime( time_corr );
722 dgf_data->SetHitPattern( HitPattern );
723 dgf_data->SetQshort( Qshort );
724 dgf_data->SetModule( mod );
725 dgf_data->SetChannel( ch );
726 dgf_data->SetEnergy( energy );
727 dgf_data->SetThreshold( thresh );
728 dgf_data->SetUserValues( UserValues );
729 dgf_data->SetUserValues( trace );
730
731 // Fill the tree
732 if( !flag_source ) {
733 std::shared_ptr<MiniballDataPackets> data_packet =
734 std::make_shared<MiniballDataPackets>( dgf_data );
735 data_vector.emplace_back( data_packet );
736 data_map.push_back( std::make_pair<unsigned long,double>(
737 data_vector.size()-1, data_packet->GetTime() ) );
738 }
739
740 }
741
742 } // channel mask, we have data in this channel
743
744 } // loop over channels
745
746 } // else of 'if(!(HitPattern&0xf) && ModuleNumber!=TS_EBIS_T1_T2_MODULE...)'
747
748 } // i < mod: data for this module
749
750 // MBS data: always 32bit words -> if len odd -> fill word has to be skipped
751 //std::cout << std::dec << "length = " << length << ", wc = " << wc << ", i = " << i << std::endl;
752 if( length & 1 ) i++;
753
754 } // data in this sub event
755
756 return;

758}

761// Function to run the conversion for a single file
762int MiniballMedConverter::ConvertFile( std::string input_file_name,
763 unsigned long start_subevt,
764 long end_subevt ) {
765
766 // Uncomment to force only a few subevts - debug
767 //end_subevt = 1000;
768
769 // Read the file.
770 std::ifstream input_file( input_file_name, std::ios::in|std::ios::binary );
771 if( !input_file.is_open() ){
772
773 std::cout << "Cannot open " << input_file_name << std::endl;
774 return -1;
775
776 }
777
778 // Reset counters to zero for every file
779 StartFile();
780
781 // Calculate the size of the file.
782 input_file.seekg( 0, input_file.end );
783 unsigned long long size_end = input_file.tellg();
784 input_file.seekg( 0, input_file.beg );
785 unsigned long long size_beg = input_file.tellg();
786 unsigned long long FILE_SIZE = size_end - size_beg;
787
788 // Calculate the number of blocks in the file.
789 unsigned long BLOCKS_NUM = FILE_SIZE / set->GetBlockSize();
790
791 std::cout << "Opening file: " << input_file_name << std::endl;
792 sslogs << "File size = " << (double)FILE_SIZE/1024./1024.;
793 sslogs << " MB" << std::endl;
794 std::cout << sslogs.str() << std::endl;
795 sslogs.str( std::string() ); // clean up
796
797 // Close the file
798 input_file.close();
799
800 // Create an MED data instance and set block/buffer size etc
801 MBS mbs;
802 mbs.SetBufferSize( set->GetBlockSize() );
803 mbs.OpenMedFile( input_file_name );
804
805 // Loop over all the MBS Events
806 unsigned long mbsevt = 0, nblock = 0;
807 for( mbsevt = 0; ; mbsevt++ ){
808
809 // Calculate how many blocks we have used and progress
810 nblock = mbs.GetBufferCount();
811 if( nblock % 200 == 0 || nblock+1 == BLOCKS_NUM ) {
812
813 // Percent complete
814 float percent = (float)(nblock+1)*100.0/(float)BLOCKS_NUM;
815
816 // Progress bar in GUI
817 if( _prog_ ){
818
819 prog->SetPosition( percent );
820 gSystem->ProcessEvents();
821
822 }
823
824 // Progress bar in terminal
825 std::cout << " " << std::setw(8) << std::setprecision(4);
826 std::cout << percent << "%\r";
827 std::cout.flush();
828
829 }
830
831 // Get the next event - returns nullptr at the end of the file and sets eof to true
832 ev = mbs.GetNextMedEvent();
833 if( mbs.IsEof() ) break;
834 if( !ev ) continue;
835 my_event_id = ev->GetEventID();
836
837 // Don't bother if we're not running
838 if( !mbs.IsRunning() ) continue;
839
840 if( my_event_id == 0 )
841 std::cout << "Bad event ID in data" << std::endl;
842
843
844 // Check if we are before the start sub event or after the end sub events
845 if( mbsevt < start_subevt || ( (long)mbsevt > end_subevt && end_subevt > 0 ) )
846 continue;
847
848 // Process current block
849 mbsinfo_packet->ClearData();
850 ProcessEvent( mbsevt );
851
852 // Write the MBS event info
853 mbsinfo_packet->SetTime( my_good_tm_stp );
854 mbsinfo_packet->SetEventID( my_event_id );
855 mbsinfo_tree->Fill();
856
857 } // loop - mbsevt < MBS_EVENTS
858
859 // Close the file
860 mbs.CloseFile();
861
862 // Print stats
863 std::cout << std::endl;
864
865 return mbsevt;
866
867}
