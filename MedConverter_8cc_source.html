78// Treat a Mesytec ADC data item
79void MiniballMedConverter::ProcessMesytecAdcData() {
80
81 // Loop over all the available data
82 unsigned int i = 0;
83 while( i < mbs_sevt->GetNumberOfData() ){
84
85 // Header of the sub event (module number in first word)
86 unsigned short header = mbs_sevt->GetData(i++);
87 unsigned short mod = header & MESYTEC_MADC_MODULE_ID;
88
89 // Convert to logical module from Marabou numbering
90 if( mod >= set->GetNumberOfAdcModules() )
91 mod = set->GetMesytecAdcModuleNumber( mod );
92 else mod--; // Mesytec modules count from 1 in Marabou!!
93
94 // Get second word of the header and test format
95 header = mbs_sevt->GetData(i++);
96 if( header & MESYTEC_MADC_OUTPUT_FORMAT ) {
97
98 std::cerr << __PRETTY_FUNCTION__ << ": Error, output ";
99 std::cerr << "format (highest bit) should be zero (header = ";
100 std::cerr << std::hex << header << std::dec << ")" << std::endl;
101 return;
102
103 }
104
105 // If format is fine, we have the word count in here
106 int wc = (header & MESYTEC_MADC_WORD_COUNT);
107
108 // Is module number ok? If not, get next item
109 if( mod < 0 ) {
110 i += wc * 2; // skip rest of event (2 shorts per word)
111 continue;
112 }
113
114 // check number of Channels (MESYTEC_MADC_NBOFCHAN=32 + 1 word End of Event + 1 extended timestamp)
115 // -> skip TOTAL subevent if wrong number of Channels
116 if( wc <= 0 || wc > (int)set->GetNumberOfMesytecAdcChannels() + 2 ) {
117
118 std::cout << __PRETTY_FUNCTION__ << ": read event nr. ";
119 std::cout << my_event_id << ": wrong Word Count: ";
120 std::cout << wc << " -> skip TOTAL subevent" << std::endl;
121 return;
122
123 }
124
125 // Loop over number of channels for which data follows (maybe also extended timestamp)
126 // Might not be all channel, so use wc, but wc includes the end of event
127 std::vector<unsigned short> ch_vec;
128 std::vector<unsigned short> qshort_vec;
129 long long Timestamp = 0;
130 bool clipped = false;
131 for( unsigned short ch = 0; ch < wc - 1; ch++ ) {
132
133 // Check if type of word (highest two bits) is end of event
134 unsigned short test = mbs_sevt->GetData(i++);
135
136 // 00 => data, 11 => end of event
137 if( ( test & MESYTEC_MADC_END_OF_EVENT ) == MESYTEC_MADC_END_OF_EVENT ) {
138
139 std::cerr << "Error, found end of event data (" << test << ") after " << ch+1;
140 std::cerr << " words but should have a word count of " << wc << std::endl;
141 return;
142
143 }
144
145 // What if only one of the higest two bits is non-zero? weird
146 else if( test & MESYTEC_MADC_END_OF_EVENT ) {
147
148 std::cerr << "Error, found weird data (" << test << ") after ";
149 std::cerr << ch+1 << " words with word count of " << wc << std::endl;
150 return;
151
152 }
153
154 // Check whether this is the extended timestamp
155 if( test & MESYTEC_MADC_EXTENDED_TIMESTAMP ) {
156
157 Timestamp = ((long long)mbs_sevt->GetData(i++)) << MESYTEC_MADC_EXTENDED_TIMESTAMP_SHIFT;
158 //Timestamp &= MESYTEC_MADC_EXTENDED_TIMESTAMP_MASK;
159 continue;
160
161 }
162
163 // If we don't have the timestamp here, test has the channel number
164 // Get actual Channel number (bits[21...16] of data word)
165 unsigned short chanNo = (test & MESYTEC_MADC_CHANNEL_NUMBER);
166 ch_vec.push_back( chanNo );
167
168 // But the energy is in the next word, lowest 12 bits
169 unsigned short Qshort = (mbs_sevt->GetData(i++) & MESYTEC_MADC_VALUE);
170 qshort_vec.push_back( Qshort );
171
172 // Check the out of range bit
173 if( mbs_sevt->GetData(i) & MESYTEC_MADC_OUT_OF_RANGE )
174 clipped = true;
175 else clipped = false;
176
177 } // loop over wc - 1
178
179 // Trailer of the sub event (module number in first word)
180 unsigned int trailer = ((unsigned int)mbs_sevt->GetData(i++) << 16) & 0xffff0000;
181 trailer |= ((unsigned int)mbs_sevt->GetData(i++)) & 0x0000ffff;
182
183 // Check type of word (highest two bits should be set)
184 if ( ( trailer & MESYTEC_MADC_END_OF_EVENT ) != MESYTEC_MADC_END_OF_EVENT ) {
185
186 std::cout << __PRETTY_FUNCTION__ << ": read event nr. " << my_event_id;
187 std::cout << ": wrong EOE word of type: " << ( trailer & MESYTEC_MADC_END_OF_EVENT );
188 std::cout << " -> skip TOTAL subevent" << std::endl;
189 return;
190
191 }
192
193 // Merge the full time stamp and add the DGF delay
194 Timestamp |= ( trailer & MESYTEC_MADC_TIMESTAMP );
195 Timestamp += set->GetDgfTimestampDelay();
196 Timestamp *= set->GetMesytecAdcTimestampUnits();
197
198 // Now we have the data, fill the tree
199 for( unsigned item = 0; item < qshort_vec.size(); item++ ){
200
201 // Clear the old stuff
202 adc_data->ClearData();
203 data_packet->ClearData();
204
205 // Some basic info for every event
206 adc_data->SetEventID( my_event_id );
207
208 // Calculate energy and threshold
209 float energy = cal->AdcEnergy( mod, ch_vec[item], qshort_vec[item] );
210 bool thresh = cal->AdcThreshold( mod, ch_vec[item] );
211
212 // Corrected time for ADCs
213 long long time_corr = Timestamp;
214 time_corr += cal->AdcTime( mod, ch_vec[item] );
215
216 // Set values for data item
217 adc_data->SetTime( time_corr ); // only works for MADC, CAEN needs reconstruction
218 adc_data->SetQshort( qshort_vec[item] );
219 adc_data->SetModule( mod );
220 adc_data->SetChannel( (char)ch_vec[item] );
221 adc_data->SetEnergy( energy );
222 adc_data->SetThreshold( thresh );
223 adc_data->SetClipped( clipped );
224
225 // Fill the tree
226 data_packet->SetData( adc_data );
227 output_tree->Fill();
228
229 // Fill histograms
230 hadc_qshort[mod][ch_vec[item]]->Fill( qshort_vec[item] );
231 hadc_cal[mod][ch_vec[item]]->Fill( energy );
232
233 }
234
235 }
236
237 return;
238
239}

355// Treat a DGF scaler data item
356void MiniballMedConverter::ProcessDgfScaler() {
357
358 // Test data comes from DGFscaler unit?
359 if( !set->CheckVmeModuleIsDgfScaler( mbs_sevt->GetModuleID() ) ){
360
361 std::cerr << __PRETTY_FUNCTION__;
362 std::cerr << ": Error - got data from VME module id ";
363 std::cerr << mbs_sevt->GetModuleID();
364 std::cerr << ", which looks like a DGF scaler module but it isn't";
365 std::cerr << " defined as such in settings file" << std::endl;
366 return;
367
368 }
369
370 // Loop over all the available data
371 unsigned int i = 0;
372 while( i < mbs_sevt->GetNumberOfData() ){
373
374 // Finish the data?
375 if( mbs_sevt->GetData(i) == DGF_SCALER_END_OF_BUFFER )
376 break;
377
378 // Check for magic word at the start
379 if( mbs_sevt->GetData(i++) != DGF_SCALER_MAGIC_WORD ){
380
381 std::cerr << "Internal dgf scalers: data out of phase - 0x";
382 std::cerr << std::hex << mbs_sevt->GetData(--i);
383 std::cerr << " (should be magic word 0x" << DGF_SCALER_MAGIC_WORD ;
384 std::cerr << ")" << std::dec << std::endl;
385 return;
386
387 }
388
389 // Get the header data
390 int wc = mbs_sevt->GetData(i++); // word count
391 int clu = mbs_sevt->GetData(i++); // cluster id
392 int mod = mbs_sevt->GetData(i++); // module id
393
394 // Break if we have nonsense
395 if( clu < 0 || mod < 0 ) return;
396 if( wc < 0 ){
397 std::cerr << "Word count negative: " << wc << std::endl;
398 return;
399 }
400 if( (int)mbs_sevt->GetNumberOfData() - (int)(i-2) < DGF_SCALER_MIN_SIZE ){
401 std::cerr << "Not enough data left in sub event for new DGF scaler: ";
402 std::cerr << mbs_sevt->GetNumberOfData() - (i-2) << " < minsize(";
403 std::cerr << DGF_SCALER_MIN_SIZE << ")" << std::endl;
404 return;
405 }
406 if( (int)mbs_sevt->GetNumberOfData() - (int)(i-2) < wc ){
407 std::cerr << "Not enough data left in sub event for new DGF scaler: ";
408 std::cerr << mbs_sevt->GetNumberOfData() - (i-2) << " < wc(" << wc << ")" << std::endl;
409 return;
410 }
411
412 // Get the real time data
413 unsigned int index = i + DGF_SCALER_INDEX_REALTIME;
414 long long realTime = mbs_sevt->GetData(index);
415 realTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
416 realTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
417
418 // Get the run time data
419 index = i + DGF_SCALER_INDEX_RUNTIME;
420 long long runTime = mbs_sevt->GetData(index);
421 runTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
422 runTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
423
424 // Get the GSLT time data
425 index = i + DGF_SCALER_INDEX_GSLTTIME;
426 long long gsltTime = mbs_sevt->GetData(index);
427 gsltTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
428 gsltTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
429
430 // Get the number of events
431 index = i + DGF_SCALER_INDEX_NEVENTS;
432 long long nEvents = mbs_sevt->GetData(index);
433 nEvents |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
434 nEvents |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
435
436 // Make a DGF scaler event and add it to the packets
437 DgfScalerData s( set->GetNumberOfDgfChannels() );
438 s.SetModule( mod );
439 s.SetClusterID( clu );
440 s.SetRealTime( realTime );
441 s.SetRunTime( runTime );
442 s.SetGSLTTime( gsltTime );
443
444 // Get channel by channel data
445 for( unsigned int j = 0; j < set->GetNumberOfDgfChannels(); j++ ){
446
447 index = i + DGF_SCALER_INDEX_CH_OFFSET;
448 index += j * DGF_SCALER_INDEX_CH_SIZE;
449
450 // Get LiveTime for this channel
451 long long liveTime = mbs_sevt->GetData(index);
452 liveTime |= ( (unsigned long long)mbs_sevt->GetData(index+1) ) << 16;
453 liveTime |= ( (unsigned long long)mbs_sevt->GetData(index+2) ) << 32;
454
455 // Get FastPeak for this channel
456 int fastPeak = mbs_sevt->GetData(index+3);
457 fastPeak |= ( (unsigned long long)mbs_sevt->GetData(index+4) ) << 16;
458
459 s.SetLiveTime( j, liveTime );
460 s.SetFastPeak( j, fastPeak );
461
462 }
463
464 // Move forward to next data item
465 i += wc - 4; // 4 header words included in word count
466
467 }
468
469 return;
470

477

479 if( mbs_sevt->GetSubEventType() != MBS_STYPE_CAMAC_DGF_3 ){
480

482 std::cerr << ": Error - got data from VME module id ";
483 std::cerr << mbs_sevt->GetModuleID();
484 std::cerr << ", which looks like a DGF module but it isn't";
485 std::cerr << " defined as such in settings file" << std::endl;

491 int i = 0;
492 int wc = mbs_sevt->GetNumberOfData();

494
495 //std::cout << std::dec << "Readout nr.: " << ev->GetEventID()-2586617;
496 //std::cout << ", wc = " << wc << ", length = " << mbs_sevt->GetDataLength();
497 //std::cout << std::hex << ", DGF data(0) = 0x" << mbs_sevt->GetData(0) << std::endl;
498
499 // Header of the sub event

501 unsigned short length = mbs_sevt->GetData(i++);
502 unsigned short end = start + length;
503 unsigned short mod = mbs_sevt->GetData(i++);
504
505 // Check length
506 if( end > mbs_sevt->GetDataLength() ){
507
508 std::cout << __PRETTY_FUNCTION__ << ": XIA wrong buffer length: ";
509 std::cout << std::dec << length << " vs. " << mbs_sevt->GetDataLength() * sizeof(unsigned short);
510 std::cout << ", Start: " << start;
511 std::cout << ", End: " << end << std::endl;
512 return; // skip total subevent
513
514 }
515
516 // Get buffer format descriptor (=RUNTASK)
517 unsigned short format = mbs_sevt->GetData(i++);
518
519 // Parameters for holding the time
520 unsigned short RunTimeA, RunTimeB, RunTimeC;
521 unsigned int BufferTime;
522 unsigned long long RunTime;
523
524 // Check if known buffer format
525 if( ( format != STD_LM_BUFFORMAT ) &&
526 ( format != COMP_LM_BUFFORMAT ) &&
527 ( format != COMP_FLM_BUFFORMAT ) &&
528 ( format != COMP3_LM_BUFFORMAT ) &&
529 ( format != STD_FLM_BUFFORMAT ) ) {
530
531 std::cout << __PRETTY_FUNCTION__ << ": read out event ";
532 std::cout << my_event_id << ": wrong buffer format: ";
533 std::cout << format << " !!!" << std::endl;
534 return; // skip total subevent
535
536 }

539
540 // read words 3-5 of buffer header: 3.: high-, 4.: middle-, 5.: low-word of run start time
541 RunTimeA = mbs_sevt->GetData(i++); // high
542 RunTimeB = mbs_sevt->GetData(i++); // mid
543 RunTimeC = mbs_sevt->GetData(i++); // low
544
545 // set 'buffertime' and 'runtime'
546 RunTime = ( (unsigned long long)RunTimeA << 32) & 0xffff000000000000;
547 BufferTime = ( (unsigned int)RunTimeB << 16) & 0xffff0000;
548 BufferTime |= ( (unsigned int)RunTimeC ) & 0x0000ffff;
549 RunTime |= ( BufferTime & 0xffffffff);
550
551 // Update timestamp using a DGF module
552 if( my_good_tm_stp == 0 )
553 my_good_tm_stp = RunTime * set->GetDgfTimestampUnits();
554 //my_good_tm_stp = RunTime;
555
556 }
557
558 // from here on: always check buffer format (=RUNTASK) before filling 'dgf' or 'dgf_rt259'
559 // beam dump module has format 'COMP3_LM_BUFFORMAT'
560 // determine corrected module number
561 mod = set->GetDgfModuleNumber( mod );
562 if( mod < 0 ) return;
563
564 // Work out what type of DGF event we have
565 int DgfType = mbs_sevt->GetSubEventType() - XIA_EVENT;
566 (void)DgfType; // not yet used anywhere
567
568 // Get data for this module
569 while( i < end ){
570
571 // Read 3 words of event header
572 // 1.: hitpattern, 2.: high-, 3.: low-word of event time
573 unsigned short HitPattern = mbs_sevt->GetData(i++);
574 unsigned short EventTimeHigh = mbs_sevt->GetData(i++);
575 unsigned short EventTimeLow = mbs_sevt->GetData(i++);
576
577 // Set 'eventtime'
578 long long EventTime = (( (long long)EventTimeHigh ) << 16) & 0xffff0000;
579 EventTime |= (long long)EventTimeLow & 0x0000ffff;
580
581 // Check for overflow and build full event time
582 if( EventTime <= BufferTime )
583 EventTime |= (( (long long)RunTimeA ) << 32) & 0xffff00000000;
584 else EventTime |= (( (long long)(RunTimeA+1) ) << 32) & 0xffff00000000;
585
586
587 // check hitpattern: at least one channel bit has to be set
588 // for all but TS_EBIS_T1_T2_MODULE and 4 CD TS-modules
589 // _all_ timestamp modules with RUNTASK!=259 -> 'ModuleNumber'
590 // can be directly compared with 'analysis module number'
591 if( !( HitPattern & 0xf ) && !set->IsTimestampModule(mod) ) {
592
593 std::cout << __PRETTY_FUNCTION__ << ": XIA hitpattern error: hitpattern = ";
594 std::cout << HitPattern << mod << ": module: " << mod;
595 std::cout << ", i: " << i << ", end: " << end;
596 std::cout << ", Data in SubEvent: " << mbs_sevt->GetNumberOfData();
597 std::cout << std::endl;
598
599 // skip module, but not full sub event
600 i = end;
601 break;
602
603 }
604
605 else {
606
607 // Process hit for all channels
608 for( unsigned int ch = 0; ch < set->GetNumberOfDgfChannels(); ch++ ) {
609
610 // Channel mask to make sure it's got data
611 unsigned int ChannelMask = 1 << ch;
612 if( ( HitPattern & ChannelMask ) != 0 ) {
613
614 //std::cout << std::hex << HitPattern << " & ";
615 //std::cout << ChannelMask << " = " << ( HitPattern & ChannelMask );
616 //std::cout << std::dec << std::endl;
617
618 // Now different data following for RUNTASK=259 resp. others
619 // RUNTASK!=259: now #of words for this channel
620 unsigned short ChannelLength;
621 if( format != COMP3_LM_BUFFORMAT )
622 ChannelLength = mbs_sevt->GetData(i++);
623
624 // Next 2 words: fast trigger time & energy for ALL diff. RUNTASKs
625 unsigned short FastTriggerTime = mbs_sevt->GetData(i++);
626 unsigned short Qshort = mbs_sevt->GetData(i++);
627
628 // Sort out long fast trigger time and wrap around
629 unsigned long long LongFastTriggerTime = FastTriggerTime;
630 if( FastTriggerTime > EventTimeLow )
631 LongFastTriggerTime += 65536ll*EventTimeHigh + 65536ll*65536ll*RunTimeA;
632 else
633 LongFastTriggerTime += 65536ll*EventTimeHigh + 65536ll + 65536ll*65536ll*RunTimeA;
634
635 // Get calibrated energy and check threshold
636 float energy = cal->DgfEnergy( mod, ch, Qshort );
637 bool thresh = cal->DgfThreshold( mod, ch );
638

640 std::vector<unsigned short> UserValues;
641 std::vector<unsigned short> trace;
642 if( format != COMP3_LM_BUFFORMAT ) {
643

645 for( unsigned char j = 0; j < 6; j++ )
646 UserValues.push_back( mbs_sevt->GetData(i++) );

648 // Read out trace
649 unsigned int TraceLength = (int)ChannelLength - CHANHEADLEN;
650 for( unsigned char j = 0; j < TraceLength; j++ )
651 trace.push_back( mbs_sevt->GetData(i++) );
652
653 } // check format for PSA and trace
654
655 //-----------
656 // Now let's add the data
657 //------------
658
659 // Fill histograms
660 hdgf_qshort[mod][ch]->Fill( Qshort );
661 hdgf_cal[mod][ch]->Fill( energy );
662
663 // Clear the old stuff
664 dgf_data->ClearData();
665 info_data->ClearData();
666 data_packet->ClearData();
667
668 //std::cout << "DGF module " << mod << ", time = ";
669 //std::cout << LongFastTriggerTime << ", Qshort = ";
670 //std::cout << Qshort << std::endl;
671
672 long long time_corr = LongFastTriggerTime * set->GetDgfTimestampUnits();
673 time_corr += cal->DgfTime( mod, ch );
674
675 // Check if it's a timestamper!
676 if( set->IsTimestampModule( mod ) ) {
677
678 // Check if it matches EBIS, T1 or SC
679 char mycode = -1;
680 if( mod == set->GetEBISDgf() && ch == set->GetEBISChannel() )
681 mycode = set->GetEBISCode();
682 else if( mod == set->GetT1Dgf() && ch == set->GetT1Channel() )
683 mycode = set->GetT1Code();
684 else if( mod == set->GetSCDgf() && ch == set->GetSCChannel() )
685 mycode = set->GetSCCode();
686
687 // Useful only if it matched
688 if( mycode >= 0 ) {
689
690 // Set values for data item
691 info_data->SetEventID( my_event_id );
692 info_data->SetTime( time_corr );
693 info_data->SetCode( mycode );
694 info_data->SetBoard( mod );
695
696 // Fill the tree
697 data_packet->SetData( info_data );
698 output_tree->Fill();
699
700 }
701
702 }
703
704 else {
705
706 // Set values for data item
707 dgf_data->SetEventID( my_event_id );
708 dgf_data->SetRunTime( RunTime * set->GetDgfTimestampUnits() );
709 dgf_data->SetEventTime( EventTime * set->GetDgfTimestampUnits() );
710 dgf_data->SetFastTriggerTime( FastTriggerTime * set->GetDgfTimestampUnits() );
711 dgf_data->SetLongFastTriggerTime( time_corr );
712 dgf_data->SetHitPattern( HitPattern );
713 dgf_data->SetQshort( Qshort );
714 dgf_data->SetModule( mod );
715 dgf_data->SetChannel( ch );
716 dgf_data->SetEnergy( energy );
717 dgf_data->SetThreshold( thresh );
718 dgf_data->SetUserValues( UserValues );
719 dgf_data->SetUserValues( trace );
720
721 // Fill the tree
722 data_packet->SetData( dgf_data );
723 output_tree->Fill();
724
725 }
726
727 } // channel mask, we have data in this channel
728
729 } // loop over channels
730
731 } // else of 'if(!(HitPattern&0xf) && ModuleNumber!=TS_EBIS_T1_T2_MODULE...)'
732
733 } // i < mod: data for this module
734
735 // MBS data: always 32bit words -> if len odd -> fill word has to be skipped
736 //std::cout << std::dec << "length = " << length << ", wc = " << wc << ", i = " << i << std::endl;
737 if( length & 1 ) i++;
738
739 } // data in this sub event
740
741 return;

746// Function to run the conversion for a single file
747int MiniballMedConverter::ConvertFile( std::string input_file_name,
748 unsigned long start_subevt,
749 long end_subevt ) {
750
751 // Uncomment to force only a few subevts - debug
752 //end_subevt = 1000;
753
754 // Read the file.
755 std::ifstream input_file( input_file_name, std::ios::in|std::ios::binary );
756 if( !input_file.is_open() ){
757
758 std::cout << "Cannot open " << input_file_name << std::endl;
759 return -1;
760
761 }
762
763 // Reset counters to zero for every file
764 StartFile();
765
766 // Calculate the size of the file.
767 input_file.seekg( 0, input_file.end );
768 unsigned long long size_end = input_file.tellg();
769 input_file.seekg( 0, input_file.beg );
770 unsigned long long size_beg = input_file.tellg();
771 unsigned long long FILE_SIZE = size_end - size_beg;
772
773 // Calculate the number of blocks in the file.
774 unsigned long BLOCKS_NUM = FILE_SIZE / set->GetBlockSize();
775
776 std::cout << "Opening file: " << input_file_name << std::endl;
777 sslogs << "File size = " << (double)FILE_SIZE/1024./1024.;
778 sslogs << " MB" << std::endl;
779 std::cout << sslogs.str() << std::endl;
780 sslogs.str( std::string() ); // clean up
781
782 // Close the file
783 input_file.close();
784
785 // Create an MED data instance and set block/buffer size etc
786 MBS mbs;
787 mbs.SetBufferSize( set->GetBlockSize() );
788 mbs.OpenMedFile( input_file_name );
789
790 // Loop over all the MBS Events
791 unsigned long mbsevt = 0, nblock = 0;
792 for( mbsevt = 0; ; mbsevt++ ){
793
794 // Calculate how many blocks we have used and progress
795 nblock = mbs.GetBufferCount();
796 if( nblock % 200 == 0 || nblock+1 == BLOCKS_NUM ) {
797
798 // Percent complete
799 float percent = (float)(nblock+1)*100.0/(float)BLOCKS_NUM;
800
801 // Progress bar in GUI
802 if( _prog_ ){
803
804 prog->SetPosition( percent );
805 gSystem->ProcessEvents();
806
807 }
808
809 // Progress bar in terminal
810 std::cout << " " << std::setw(8) << std::setprecision(4);
811 std::cout << percent << "%\r";
812 std::cout.flush();
813
814 }
815
816 // Get the next event - returns nullptr at the end of the file and sets eof to true
817 ev = mbs.GetNextMedEvent();
818 if( mbs.IsEof() ) break;
819 if( !ev ) continue;
820 my_event_id = ev->GetEventID();
821
822 // Don't bother if we're not running
823 if( !mbs.IsRunning() ) continue;
824
825 if( my_event_id == 0 )
826 std::cout << "Bad event ID in data" << std::endl;
827
828
829 // Check if we are before the start sub event or after the end sub events
830 if( mbsevt < start_subevt || ( (long)mbsevt > end_subevt && end_subevt > 0 ) )
831 continue;
832
833 // Process current block
834 mbsinfo_packet->ClearData();
835 ProcessEvent( mbsevt );
836
837 // Write the MBS event info
838 mbsinfo_packet->SetTime( my_good_tm_stp );
839 mbsinfo_packet->SetEventID( my_event_id );
840 mbsinfo_tree->Fill();
841
842 } // loop - mbsevt < MBS_EVENTS
843
844 // Close the file
845 mbs.CloseFile();
846
847 // Print stats
848 std::cout << std::endl;
849
850 return mbsevt;
851
852}
