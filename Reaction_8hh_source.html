319 // Reaction calculations
320 inline double GetQvalue(){
321 return Beam.GetMass() + Target.GetMass() -
322 Ejectile.GetMass() - Recoil.GetMass();
323 };
324 inline double GetEnergyTotLab(){
325 return Beam.GetEnergyTot() + Target.GetEnergyTot();
326 };
327 inline double GetEnergyTotCM(){
328 double etot = TMath::Power( Beam.GetMass(), 2.0 );
329 etot += TMath::Power( Target.GetMass(), 2.0 );
330 etot += 2.0 * Beam.GetEnergyTot() * Target.GetMass();
331 etot = TMath::Sqrt( etot );
332 return etot;
333 };
334 inline double GetBeta(){
335 double beta2 = 0.25 * Beam.GetMass() + 1.5 * Beam.GetEnergy();
336 beta2 = TMath::Sqrt( beta2 * Beam.GetMass() );
337 beta2 -= 0.5 * Beam.GetMass();
338 return TMath::Sqrt( beta2 / ( 0.75 * Beam.GetMass() ) );
339 //return TMath::Sqrt( 2.0 * Beam.GetEnergy() / Beam.GetMass() );
340 };
341 inline double GetGamma(){
342 return 1.0 / TMath::Sqrt( 1.0 - TMath::Power( GetBeta(), 2.0 ) );
343 };
344 inline double GetTau(){
345 return Beam.GetMass() / Target.GetMass();
346 };
347 inline double GetEnergyPrime(){
348 return Beam.GetEnergy() - ( Ejectile.GetEx() + Recoil.GetEx() ) * ( 1.0 + GetTau() );
349 };
350 inline double GetEpsilon(){
351 return TMath::Sqrt( Beam.GetEnergy() / GetEnergyPrime() );
352 };
