319 inline double GetQvalue(){
320 return Beam.GetMass() + Target.GetMass() -
321 Ejectile.GetMass() - Recoil.GetMass();
322 };
323 inline double GetEnergyTotLab(){
324 return Beam.GetEnergyTot() + Target.GetEnergyTot();
325 };
326 inline double GetEnergyTotCM(){
327 double etot = TMath::Power( Beam.GetMass(), 2.0 );
328 etot += TMath::Power( Target.GetMass(), 2.0 );
329 etot += 2.0 * Beam.GetEnergyTot() * Target.GetMass();
330 etot = TMath::Sqrt( etot );
331 return etot;
332 };
333 inline double GetBeta(){
334 double beta2 = 0.25 * Beam.GetMass() + 1.5 * Beam.GetEnergy();
335 beta2 = TMath::Sqrt( beta2 * Beam.GetMass() );
336 beta2 -= 0.5 * Beam.GetMass();
337 return TMath::Sqrt( beta2 / ( 0.75 * Beam.GetMass() ) );
338 //return TMath::Sqrt( 2.0 * Beam.GetEnergy() / Beam.GetMass() );
339 };
340 inline double GetGamma(){
341 return 1.0 / TMath::Sqrt( 1.0 - TMath::Power( GetBeta(), 2.0 ) );
342 };
343 inline double GetTau(){
344 return Beam.GetMass() / Target.GetMass();
345 };
346 inline double GetEnergyPrime(){
347 return Beam.GetEnergy() - ( Ejectile.GetEx() + Recoil.GetEx() ) * ( 1.0 + GetTau() );
348 };
349 inline double GetEpsilon(){
350 return TMath::Sqrt( Beam.GetEnergy() / GetEnergyPrime() );
351 };
