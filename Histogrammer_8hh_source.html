224 // Particle energy vs angle cuts
225 inline bool TransferCut( std::shared_ptr<ParticleEvt> p ){
226 double xval = p->GetEnergy();
227 double yval = p->GetDeltaEnergy();
228 if( react->GetTransferX() == "dE" ) xval = p->GetDeltaEnergy();
229 else if( react->GetTransferX() == "E" ) xval = p->GetEnergy();
230 else if( react->GetTransferX() == "theta" ) xval = react->GetParticleTheta(p) * TMath::RadToDeg();
231 if( react->GetTransferY() == "dE" ) yval = p->GetDeltaEnergy();
232 else if( react->GetTransferY() == "E" ) yval = p->GetEnergy();
233 else if( react->GetTransferY() == "theta" ) yval = react->GetParticleTheta(p) * TMath::RadToDeg();
234 return react->GetTransferCut()->IsInside( xval , yval );
235 }

242 inline bool TwoParticleCut( std::shared_ptr<ParticleEvt> p1, std::shared_ptr<ParticleEvt> p2 ){
243 if( EjectileCut(p1) && RecoilCut(p2) && PromptCoincidence( p1, p2 ) &&
244 TMath::Abs( react->GetParticleVector(p1).DeltaPhi( react->GetParticleVector(p2) ) ) < 1.5*TMath::Pi() &&
245 TMath::Abs( react->GetParticleVector(p1).DeltaPhi( react->GetParticleVector(p2) ) ) > 0.5*TMath::Pi() )
246 return true;
