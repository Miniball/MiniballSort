159// Function to call the monitoring loop
160void* monitor_run( void* ptr ){
161
162 // This doesn't make sense for MED data which is historical
163 if( flag_med ) return 0;
164
165 // Get the settings, file etc.
166 thptr *calfiles = (thptr*)ptr;
167
168 // Load macros in thread
169 std::string rootline = ".L " + std::string(CUR_DIR) + "include/MonitorMacros.hh";
170 gROOT->ProcessLine( rootline.data() );
171
172 // This function is called to run when monitoring
173 if( flag_mbs ){
174 conv_mbs_mon = std::make_shared<MiniballMbsConverter>( calfiles->myset );
175 conv_mon.reset( conv_mbs_mon.get() );
176 }
177 else if( flag_midas ) {
178 conv_midas_mon = std::make_shared<MiniballMidasConverter>( calfiles->myset );
179 conv_mon.reset( conv_midas_mon.get() );
180 }
181 eb_mon = std::make_shared<MiniballEventBuilder>( calfiles->myset );
182 hist_mon = std::make_shared<MiniballHistogrammer>( calfiles->myreact, calfiles->myset );
183
184
185 // Data blocks for Data spy
186 if( flag_spy && ( myset->GetBlockSize() != 0x10000 && flag_midas ) ) {
187
188 // only 64 kB supported atm
189 std::cerr << "Currently only supporting 64 kB block size" << std::endl;
190 exit(1);
191
192 }
193
194 // Daresbury MIDAS DataSpy
195 DataSpy myspy;
196 long long buffer[8*1024];
197 int file_id = 0;
198 if( flag_spy && flag_midas ) myspy.Open( file_id );
199 int spy_length = 0;
200
201 // GSI MBS EventServer
202 MBS mbs;
203 if( flag_spy && flag_mbs ) mbs.OpenEventServer( "localhost", 8030 );
204
205 // Data/Event counters
206 int start_block = 0, start_subevt = 0;
207 int nblocks = 0, nsubevts = 0;
208 unsigned long nbuild = 0;
209
210 // Converter setup
211 if( !flag_spy ) curFileMon = input_names.at(0); // maybe change in GUI later?
212 if( flag_source ) conv_mon->SourceOnly();
213 if( flag_ebis ) conv_mon->EBISOnly();
214 conv_mon->AddCalibration( calfiles->mycal );
215 conv_mon->SetOutput( "monitor_singles.root" );
216 conv_mon->MakeTree();
217 conv_mon->MakeHists();
218
219 // Update server settings
220 // title of web page
221 std::string toptitle;
222 if( !flag_spy ) toptitle = curFileMon.substr( curFileMon.find_last_of("/")+1,
223 curFileMon.length()-curFileMon.find_last_of("/")-1 );
224 else toptitle = "DataSpy ";
225 toptitle += " (" + std::to_string( mon_time ) + " s)";
226 serv->SetItemField("/", "_toptitle", toptitle.data() );
227
228 // While the sort is running
229 while( true ) {
230
231 // bRunMon can be set by the GUI
232 while( bRunMon ) {
233
234 // Convert - from MIDAS file
235 if( !flag_spy && flag_midas ) {
236
237 nblocks = conv_midas_mon->ConvertFile( curFileMon, start_block );
238 start_block = nblocks;
239
240 }
241
242 // Convert - from MBS file
243 else if( !flag_spy && flag_mbs ) {
244
245 nsubevts = conv_mbs_mon->ConvertFile( curFileMon, start_subevt );
246 start_subevt = nsubevts;
247
248 }
249
250 // Convert - from MIDAS shared memory
251 else if( flag_spy && flag_midas ){
252
253 // Clean up the trees before we start
254 conv_midas_mon->GetSortedTree()->Reset();
255 conv_midas_mon->GetMbsInfo()->Reset();
256
257 // First check if we have data
258 std::cout << "Looking for data from DataSpy" << std::endl;
259 spy_length = myspy.Read( file_id, (char*)buffer, calfiles->myset->GetBlockSize() );
260 if( spy_length == 0 && bFirstRun ) {
261 std::cout << "No data yet on first pass" << std::endl;
262 gSystem->Sleep( 2e3 );
263 continue;
264 }
265
266 // Keep reading until we have all the data
267 // This could be multi-threaded to process data and go back to read more
268 int block_ctr = 0;
269 long byte_ctr = 0;
270 int poll_ctr = 0;
271 while( block_ctr < 200 && poll_ctr < 1000 ){
272
273 //std::cout << "Got some data from DataSpy" << std::endl;
274 if( spy_length > 0 ) {
275 nblocks = conv_midas_mon->ConvertBlock( (char*)buffer, 0 );
276 block_ctr += nblocks;
277 }
278
279 // Read a new block
280 gSystem->Sleep( 1 ); // wait 1 ms between each read
281 spy_length = myspy.Read( file_id, (char*)buffer, calfiles->myset->GetBlockSize() );
282
283 byte_ctr += spy_length;
284 poll_ctr++;
285
286 }
287
288 std::cout << "Got " << byte_ctr << " bytes of data from DataSpy" << std::endl;
289
290 // Sort the packets we just got, then do the rest of the analysis
291 conv_midas_mon->SortTree();
292 conv_midas_mon->PurgeOutput();
293
294 }
295
296 // Convert - from MBS event server
297 else if( flag_spy && flag_mbs ){
298
299 // First check if we have data
300 std::cout << "Looking for data from MBSEventServer" << std::endl;
301 conv_mbs_mon->SetMBSEvent( mbs.GetNextEventFromStream() );
302 conv_mbs_mon->ProcessBlock(0);
303 conv_mbs_mon->SortTree();
304 conv_mbs_mon->PurgeOutput();
305
306 }
307
308
309 // Only do the rest if it is not a source run
310 if( !flag_source ) {
311
312 // Event builder
313 if( bFirstRun ) {
314 eb_mon->SetOutput( "monitor_events.root" );
315 eb_mon->StartFile();
316
317 }
318 // TODO: This could be done better with smart pointers
319 TTree *sorted_tree = conv_mon->GetSortedTree()->CloneTree();
320 TTree *mbsinfo_tree = conv_mon->GetMbsInfo()->CloneTree();
321 eb_mon->SetInputTree( sorted_tree );
322 eb_mon->SetMBSInfoTree( mbsinfo_tree );
323 eb_mon->GetTree()->Reset();
324 nbuild = eb_mon->BuildEvents();
325 eb_mon->PurgeOutput();
326 delete sorted_tree;
327 delete mbsinfo_tree;
328
329 // Histogrammer
330 if( bFirstRun ) {
331 hist_mon->SetOutput( "monitor_hists.root" );
332 }
333 if( nbuild ) {
334 // TODO: This could be done better with smart pointers
335 TTree *evt_tree = eb_mon->GetTree()->CloneTree();
336 hist_mon->SetInputTree( evt_tree );
337 hist_mon->FillHists();
338 hist_mon->PurgeOutput();
339 delete evt_tree;
340 }
341
342 // If this was the first time we ran, do stuff?
343 if( bFirstRun ) {
344
345 bFirstRun = kFALSE;
346
347 }
348
349 }
350
351 // This makes things unresponsive!
352 // Unless we are threading?
353 gSystem->Sleep( mon_time * 1e3 );
354
355 } // bRunMon
356
357 } // always running
358
359 // Close the dataSpy before exiting (no point really)
360 if( flag_spy && flag_midas ) myspy.Close( file_id );
361 if( flag_spy && flag_mbs ) mbs.CloseEventServer();
362
363 // Close all outputs (we never reach here anyway)
364 conv_mon->CloseOutput();
365 eb_mon->CloseOutput();
366 hist_mon->CloseOutput();
367
368 return 0;
369
370}

406 // Run conversion to ROOT //

409 MiniballMidasConverter conv_midas( myset );
410 MiniballMbsConverter conv_mbs( myset );
411 MiniballMedConverter conv_med( myset );
412 std::cout << "\n +++ Miniball Analysis:: processing MiniballConverter +++" << std::endl;
413
414 TFile *rtest;
415 std::ifstream ftest;
416 std::string name_input_file;
417 std::string name_output_file;
418
419 // Check each file
420 for( unsigned int i = 0; i < input_names.size(); i++ ){
421
422 name_input_file = input_names.at(i).substr( input_names.at(i).find_last_of("/")+1,
423 input_names.at(i).length() - input_names.at(i).find_last_of("/")-1 );
424 name_input_file = name_input_file.substr( 0, name_input_file.find_last_of(".") );
425
426 if( flag_source ) name_output_file = name_input_file + "_source.root";
427 else name_output_file = name_input_file + ".root";
428
429 name_output_file = datadir_name + "/" + name_output_file;
430 name_input_file = input_names.at(i);
431
432 force_convert.push_back( false );
433
434 // If input doesn't exist, skip it
435 ftest.open( name_input_file.data() );
436 if( !ftest.is_open() ) {
437
438 std::cerr << name_input_file << " does not exist" << std::endl;
439 continue;
440
441 }
442 else ftest.close();
443
444 // If output doesn't exist, we have to convert it anyway
445 // The convert flag will force it to be converted
446 ftest.open( name_output_file.data() );
447 if( !ftest.is_open() ) force_convert.at(i) = true;
448 else {
449
450 ftest.close();
451 rtest = new TFile( name_output_file.data() );
452 if( rtest->IsZombie() ) force_convert.at(i) = true;
453 if( !flag_convert && !force_convert.at(i) )
454 std::cout << name_output_file << " already converted" << std::endl;
455 rtest->Close();
456
457 }
458
459 if( flag_convert || force_convert.at(i) ) {
460
461 std::cout << name_input_file << " --> ";
462 std::cout << name_output_file << std::endl;

471 conv_mbs.MakeHists();
472 conv_mbs.ConvertFile( name_input_file );
473
474 // Sort the tree before writing and closing
475 if( !flag_source ){
476 conv_mbs.BuildMbsIndex();
477 if( myset->GetMbsEventMode() )
478 conv_mbs.NoSortTree();
479 else conv_mbs.SortTree();
480 }
481 conv_mbs.CloseOutput();
482
483 }
484
485 else if( flag_midas ) {
486
487 if( flag_source ) conv_midas.SourceOnly();
488 if( flag_ebis ) conv_midas.EBISOnly();
489 conv_midas.SetOutput( name_output_file );
490 conv_midas.AddCalibration( mycal );
491 conv_midas.MakeTree();
492 conv_midas.MakeHists();
493 conv_midas.ConvertFile( name_input_file );
494
495 // Sort the tree before writing and closing
496 if( !flag_source ) {
497 conv_midas.SortTree();
498 //conv_midas.BodgeMidasSort();
499 }
500 conv_midas.CloseOutput();
501
502 }
503
504 else if( flag_med ){
505
506 if( flag_source ) conv_med.SourceOnly();
507 if( flag_ebis ) conv_med.EBISOnly();
508 conv_med.SetOutput( name_output_file );
509 conv_med.AddCalibration( mycal );
510 conv_med.MakeTree();
511 conv_med.MakeHists();
512 conv_med.ConvertFile( name_input_file );
513
514 // Sort the tree before writing and closing
515 if( !flag_source ){
516 conv_med.BuildMbsIndex();
517 if( myset->GetMbsEventMode() )
518 conv_med.NoSortTree();
519 else conv_med.SortTree();
520 }
521 conv_med.CloseOutput();
522
523 }
524
525 }
526
527 }

529 return;

561 ftest.open( name_input_file.data() );

563

566
567 }
568 else {
569
570 ftest.close();
571 return_flag = true;
572
573 }
574
575 // We need to do event builder if we just converted it
576 // specific request to do new event build with -e
577 // this is useful if you need to add a new calibration
578 if( flag_convert || force_convert.at(i) || flag_events )
579 force_events = true;
580
581 // If it doesn't exist, we have to sort it anyway
582 else {
583
584 ftest.open( name_output_file.data() );
585 if( !ftest.is_open() ) force_events = true;
586 else {
587
588 ftest.close();
589 rtest = new TFile( name_output_file.data() );
590 if( rtest->IsZombie() ) force_events = true;
591 if( !force_events )
592 std::cout << name_output_file << " already built" << std::endl;
593 rtest->Close();
594
595 }
596
597 }
598
599 if( force_events ) {
600
601 std::cout << name_input_file << " --> ";
602 std::cout << name_output_file << std::endl;
603
604 eb.SetInputFile( name_input_file );
605 eb.SetOutput( name_output_file );
606 eb.BuildEvents();
607 eb.CloseOutput();
608
609 force_events = false;
610
611 }
612
613 }
614
615 return return_flag;

668void do_angle_fit(){
669
670 //------------------------------------------//
671 // Run angle fitting routine with 22Ne data //
672 //------------------------------------------//
673 MiniballAngleFitter angle_fit( myset, myreact );
674 std::cout << "\n +++ Miniball Analysis:: processing MiniballAngleFitter +++" << std::endl;
675
676 TFile *rtest;
677 std::ifstream ftest;
678 std::string name_input_file;
679 std::string name_output_file = "22Ne_angle_fit.root";
680 std::string hadd_file_list = "";
681 std::string name_results_file = "22Ne_angle_fit.cal";
682
683 // Check each file
684 for( unsigned int i = 0; i < input_names.size(); i++ ){
685
686 name_input_file = input_names.at(i).substr( input_names.at(i).find_last_of("/")+1,
687 input_names.at(i).length() - input_names.at(i).find_last_of("/")-1 );
688 name_input_file = name_input_file.substr( 0,
689 name_input_file.find_last_of(".") );
690 name_input_file = datadir_name + "/" + name_input_file + "_events.root";
691
692 // Add to list if the converted file exists
693 ftest.open( name_input_file.data() );
694 if( ftest.is_open() ) {
695
696 ftest.close();
697 rtest = new TFile( name_input_file.data() );
698 if( !rtest->IsZombie() ) {
699 hadd_file_list += " " + name_input_file;
700 }
701 else {
702 std::cout << "Skipping " << name_input_file;
703 std::cout << ", it's broken" << std::endl;
704 }
705 rtest->Close();
706
707 }
708
709 else {
710
711 std::cout << "Skipping " << name_input_file;
712 std::cout << ", file does not exist" << std::endl;
713
714 }
715
716 }
717
718 // If we have some ROOT files, add them and pass to the fitter
719 if( input_names.size() ){
720
721 // Perform the hadd (doesn't work on Windows)
722 gErrorIgnoreLevel = kError;
723 std::string cmd = "hadd -k -T -v 0 -f ";
724 cmd += name_output_file;
725 cmd += hadd_file_list;
726 gSystem->Exec( cmd.data() );
727 gErrorIgnoreLevel = kInfo;
728
729 // Give this file to the angle fitter
730 if( !angle_fit.SetInputROOTFile( name_output_file ) ) return;
731
732 }
733
734 // Otherwise we have to take the energies from the file
735 else if( !angle_fit.SetInputEnergiesFile( name_angle_file ) ) return;
736
737 // Perform the fitting
738 angle_fit.DoFit();
739
740 // Save the experimental energies and angles to a file
741 if( input_names.size() )
742 angle_fit.SaveExpEnergies( "22Ne_fitted_energies.dat" );
743 angle_fit.SaveReactionFile( name_results_file );
744
745 // Close the ROOT file
746 angle_fit.CloseROOTFile();
747
748}
